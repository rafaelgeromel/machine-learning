# partial derivative of the cost(loss) with respect to w1 (slogpe)
def dLdw1():
    m = X.size
    return ((1/m) * np.sum(X * (f(X) - Y)))
   
    
# partial derivative of the cost(loss) with respect to w0 (slogpe)
def dLdw0():
    m = X.size
    return ((1/m) * np.sum(f(X) - Y))

    
def cost():
    m = X.size
    return ((1/(2 * m)) * (np.sum(Y - f(X)) ** 2))
